cmake_minimum_required(VERSION 3.10)

project(OX_GAME)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# output
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/Debug/bin)
set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/Debug/lib)
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/Debug/lib)

# library
add_library(game_lib STATIC
    src/Game.cpp
    src/Board.cpp
    src/Player.cpp
)

target_include_directories(game_lib PUBLIC include)

# executable
add_executable(OX_GAME main.cpp)

target_link_libraries(OX_GAME PRIVATE game_lib)
